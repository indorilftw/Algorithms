\begin{algorithm}[H]
\caption{Άσκηση 1}
\begin{algorithmic}[1]
\Procedure{Representants Selection}{$A, N$}
    \State Sort A on ascending order using f\textsubscript{i} as key
    \State $i = 0$
    \While{i < N \textbf{and} s\textsubscript{i} < f\textsubscript{1}}
        \State $i \gets i+1$
    \EndWhile
    \State $i \gets i-1$
    \State $RepresentativesList.append (i)$
    \For{j \gets i to 1 \step -1}
	A.remove (j)
    \EndFor
    \State $j \gets 1$
    \While{$\lnot \Call{Empty}{t}$}
        \State $item \gets \Call{ExtractMin}{t}$
    \EndWhile
\EndProcedure
\end{algorithmic}
\end{algorithm}
